## 一、功能概述

### 1.1 系统架构

Digital Labor 是一套面向建筑劳务行业的数字化管理系统，采用前后端分离架构：

```
┌─────────────────────────────────────────────────────────────┐
│                    Digital Labor 系统                      │
├─────────────────────────────────────────────────────────────┤
│  PC管理端 (Next.js)    │  H5工人端 (Next.js)      │
│  - 人员档案管理        │  - 个人中心               │
│  - 合同管理            │  - 我的合同               │
│  - 考勤管理            │  - 我的考勤               │
│  - 结算管理            │  - 我的薪资               │
│  - 现场管理            │  - 消息通知               │
│  - 系统管理            │  - 激活流程               │
├─────────────────────────────────────────────────────────────┤
│              后端服务 (Node.js + Express)                │
│  - 认证授权  - 业务API  - 数据处理  - 文件管理       │
├─────────────────────────────────────────────────────────────┤
│              数据库 (SQLite)  +  文件存储               │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 功能模块划分

3. 修改 server/routes/person.js:
```javascript
import axios from 'axios'

router.post('/face-verify', async (req, res) => {
  const { image, person_id } = req.body || {}
  
  if (!image && !person_id) {
    return res.status(400).json({ message: '请提供 image（base64）或 person_id' })
  }
  
  try {
    const response = await axios.post(
      `${process.env.FACE_VERIFY_ENDPOINT}/api/v1/face/verify`,
      {
        image: image,
        person_id: person_id
      },
      {
        headers: {
          'Authorization': `Bearer ${process.env.FACE_VERIFY_API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    )
    
    if (response.data.code === '200' && response.data.data.passed) {
      res.json({ ok: true, message: '核验通过' })
    } else {
      res.json({ ok: false, message: response.data.message || '核验失败' })
    }
  } catch (error) {
    console.error('人脸验证失败:', error)
    res.status(500).json({ message: '人脸验证服务异常' })
  }
})
```

```javascript
import axios from 'axios'

const ESIGN_CONFIG = {
  appId: process.env.ESIGN_APP_ID,
  appSecret: process.env.ESIGN_APP_SECRET,
  endpoint: process.env.ESIGN_ENDPOINT
}

export async function createContract(templateId, title, personId, deadline) {
  const response = await axios.post(
    `${ESIGN_CONFIG.endpoint}/v3/sign-flow/create`,
    {
      templateId: templateId,
      title: title,
      signers: [
        {
          signerAccountId: personId,
          signerType: 'PERSON',
          order: 1
        }
      ],
      deadline: deadline
    },
    {
      headers: {
        'Authorization': `Bearer ${await getAccessToken()}`,
        'Content-Type': 'application/json'
      }
    }
  )
  
  return response.data
}

export async function downloadContractPdf(flowId) {
  const response = await axios.get(
    `${ESIGN_CONFIG.endpoint}/v3/sign-flow/${flowId}/file`,
    {
      headers: {
        'Authorization': `Bearer ${await getAccessToken()}`
      },
      responseType: 'arraybuffer'
    }
  )
  
  return response.data
}

async function getAccessToken() {
  const response = await axios.post(
    `${ESIGN_CONFIG.endpoint}/oauth2/access_token`,
    {
      appId: ESIGN_CONFIG.appId,
      appSecret: ESIGN_CONFIG.appSecret
    }
  )
  
  return response.data.accessToken
}
```

```javascript
      res.sendFile(pdfPath)
    } else if (row.pdf_path) {
      const full = path.resolve(__dirname, '..', row.pdf_path)
      if (!fs.existsSync(full)) return res.status(404).json({ message: '文件不存在' })
      res.sendFile(full)
    } else {
      res.status(404).json({ message: '暂无 PDF 文件', noFile: true })
    }
  } catch (error) {
    console.error('PDF下载失败:', error)
    res.status(500).json({ message: 'PDF下载失败' })
  }
})
```

```javascript
import bcrypt from 'bcrypt'

router.post('/login', (req, res) => {
  const { username, password } = req.body || {}
  const u = (username != null && typeof username === 'string') ? username.trim() : ''
  const p = (password != null && typeof password === 'string') ? password : ''
  if (!u || !p) return err(res, 400, '用户名和密码必填')
  
  const row = db.prepare('SELECT id, username, name, role, password_hash FROM user WHERE username = ? AND enabled = 1').get(u)
  if (!row) return err(res, 401, '用户名或密码错误')
  
  const isValid = bcrypt.compareSync(p, row.password_hash)
  if (!isValid) return err(res, 401, '用户名或密码错误')
  
  const token = signToken({ userId: row.id, username: row.username })
  res.json({ token, user: { id: row.id, username: row.username, name: row.name, role: row.role } })
})
```

#### 8.2.2 本地缓存

**Node.js内存缓存**：
```javascript
// server/lib/localCache.js
class LocalCache {
  constructor() {
    this.cache = new Map()
    this.timers = new Map()
  }
  
  get(key) {
    return this.cache.get(key)
  }
  
  set(key, value, ttl = 60000) {
    this.cache.set(key, value)
    
    // 清除旧的定时器
    if (this.timers.has(key)) {
      clearTimeout(this.timers.get(key))
    }
    
    // 设置过期
    const timer = setTimeout(() => {
      this.cache.delete(key)
      this.timers.delete(key)
    }, ttl)
    
    this.timers.set(key, timer)
  }
  
  del(key) {
    this.cache.delete(key)
    if (this.timers.has(key)) {
      clearTimeout(this.timers.get(key))
      this.timers.delete(key)
    }
  }
}

export const localCache = new LocalCache()
```

### 9.2 数据安全

#### 9.2.1 敏感数据加密

```javascript
import crypto from 'crypto'

// AES加密敏感字段
const algorithm = 'aes-256-gcm'
const key = crypto.scryptSync(process.env.ENCRYPTION_KEY, 'salt', 32)

export function encrypt(text) {
  const iv = crypto.randomBytes(16)
  const cipher = crypto.createCipheriv(algorithm, key, iv)
  
  let encrypted = cipher.update(text, 'utf8', 'hex')
  encrypted += cipher.final('hex')
  
  const authTag = cipher.getAuthTag()
  
  return {
    iv: iv.toString('hex'),
    authTag: authTag.toString('hex'),
    encrypted
  }
}

export function decrypt(encryptedData) {
  const decipher = crypto.createDecipheriv(
    algorithm, 
    key, 
    Buffer.from(encryptedData.iv, 'hex')
  )
  decipher.setAuthTag(Buffer.from(encryptedData.authTag, 'hex'))
  
  let decrypted = decipher.update(encryptedData.encrypted, 'hex', 'utf8')
  decrypted += decipher.final('utf8')
  
  return decrypted
}

// 使用示例：加密身份证号
const encryptedIdCard = encrypt(person.id_card)
db.prepare('UPDATE person SET id_card_encrypted = ? WHERE id = ?')
  .run(JSON.stringify(encryptedIdCard), person.id)
```

```javascript
import multer from 'multer'

    cb(null, true)
  } else {
    cb(new Error('不支持的文件类型'), false)
  }
}

  fileFilter,
```
